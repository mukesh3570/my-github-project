#####batch class##########
global class TestDemoContactBatch implements database.Batchable<SObject>{
    global Database.QueryLocator start(database.BatchableContext bc){
        return Database.getQueryLocator([select id,LastName from Contact]);
       
    }
    
    global void execute(database.BatchableContext bc,List<Contact> sub){
        for(Contact con: sub){
            con.LastName = 'Dell!';
        }
        
        update sub;
        
        }
    global void finish(database.BatchableContext bc){}
        
    }

################scheduler class#############

global class contactBatchSchedule implements Schedulable{
    global void execute(SchedulableContext sc) {
        
        Database.executeBatch(new TestDemoContactBatch());
    }
} 


########execute scheduler class#####
System.schedule('0 0 1 * * ?', new contactBatchSchedule());
